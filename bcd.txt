.MODEL SMALL
.STACK
.DATA
PA EQU 0DC50H
CW DB 80H
CR EQU ODC53H
.CODE
MOV AX,@DATA
MOV DS,AX
MOV AL,CW
MOV DX,VR
OUT DX,AL
MOV AL,0
UP:MOV DX,PA
OUT DX,AL
CALL DELAY
ADD AL,1
DAA
CMP AL,99H
JNE UP
DOWN:MOV DX,PA
OUT DX,AL
CALL DELAY
ADD AL,99H
DAA
CMP AL,99H
JNE DOWN
MOV AH,4CH
INT 21H

DELAY PROC
MOV BX,0FFFH
B2:MOV CX,0FFFFH
B1:LOOP B1
DEC BX
JNZ B2
RET
DELAY ENDP
END
